Not: ~- 1
NoMatch: == Not
Has: ? < p#
In: ~Has
HasAny: `In \>:
AnyIn: ~HasAny

Less: P #: .(`== Not)
LessAll: P #: .(`In Not)

Drop: P # .(Q - p#)

Cut_2: { 0,(y?->is+1) '(X Drop P # ~-) (is,(y#)) }
